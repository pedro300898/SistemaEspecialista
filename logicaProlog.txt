ListaSintomas = [SintomaA, SintomaB, SintomaC, SintomaD, SintomaE, SintomaF, SintomaG, SintomaH, SintomaI].

encontraDoenca(ListaSintomas, Doenca, Lista):-
    findall(Doenca, encontraDoencaSintomaExclusivo(ListaSintomas, Doenca), Lista).

encontraDoencaSintomaExclusivo(ListaSintomas, Doenca):-
    sintoma(Doenca, ListaSintomas[0]),
    sintoma(Doenca, ListaSintomas[1]),
    sintoma(Doenca, ListaSintomas[2]),
    sintoma(Doenca, ListaSintomas[3]),
    sintoma(Doenca, ListaSintomas[4]),
    sintoma(Doenca, ListaSintomas[5]),
    sintoma(Doenca, ListaSintomas[6]),
    sintoma(Doenca, ListaSintomas[7]),
    sintoma(Doenca, ListaSintomas[8]).

retornaDoenca(Doenca, ListaSintomas, X) :- 
    sintoma([X,Doenca], ListaSintomas).
